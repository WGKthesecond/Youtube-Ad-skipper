(function(){
  if (window.__adSkipperActive) {
    console.log("Ad skipper already running!");
    return;
  }
  window.__adSkipperActive = true;

  let video = document.querySelector('video');
  if (!video) {
    alert('No video found!');
    return;
  }

  let overlay = document.createElement('div');
  overlay.textContent = 'Ad is being skipped, please hold...';
  Object.assign(overlay.style, {
    position: 'absolute',
    top: '50%',
    left: '50%',
    transform: 'translate(-50%, -50%)',
    background: 'rgba(0,0,0,0.7)',
    color: 'white',
    padding: '20px',
    fontSize: '24px',
    borderRadius: '10px',
    zIndex: '9999',
    display: 'none',
    pointerEvents: 'none',
    textAlign: 'center'
  });
  document.body.appendChild(overlay);

  function checkAd() {
    const adContainer = document.querySelector('.ad-showing');
    if (adContainer) {
      video.playbackRate = 16;
      video.muted = true;
      overlay.style.display = 'block';
      console.log('[Ad Skipper] Ad detected: Speeding up and muting...');
    } else {
      video.playbackRate = 1;
      video.muted = false;
      overlay.style.display = 'none';
      console.log('[Ad Skipper] No ad: Normal speed and unmuted.');
    }
  }

  setInterval(checkAd, 500);
})();
